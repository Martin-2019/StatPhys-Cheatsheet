\subsection*{Virial expansion}

\[
    \begin{aligned}
        F &= N k_B T \left[\ln(\rho \lambda^3)-1+B_2\rho\right] \\
        p &= \rho k_B T \left[1 + B_2 \rho\right]
    \end{aligned}
\]
