\subsection*{Stirling's formula}

\begin{equation*}
    \ln (n!) = n  \ln (n) - n + \frac{1}{2} \ln (2 \pi n)
\end{equation*}

\begin{equation*}
    \epsilon = \frac{p^2}{2m} = \frac{\hbar^2 k^2}{2m}
\end{equation*}
